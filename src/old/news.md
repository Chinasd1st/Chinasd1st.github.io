```python
def encrypt():
    temp = raw_input("Please input your sentence: ")
    key = int(raw_input("Please input your key: "))
    listA = map(ord, temp)
    lens = len(listA)
    for i in range(lens):
        a = listA[i]
        if 65 <= a <= 90:
            a += key
            while a > 90:
                a -= 26
        elif 97 <= a <= 122:
            a += key
            while a > 122:
                a -= 26
        listA[i] = a
    listA = map(chr, listA)
    listA = ''.join(listA)
```
